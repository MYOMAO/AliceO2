147 void RawReaderCRU::findSyncPositions()
148 {
149  std::ifstream file;
150  file.open(mInputFileName, std::ifstream::binary);
151  if (!file.good())
152  throw std::runtime_error("Unable to open or access file " + mInputFileName);
153 
154  // loop over the mNumberLinks potential links in the data
155  // only if data from the link is present and selected
156  // for decoding it will be decoded.
157  for (int link = 0; link < mNumberLinks; link++) {
158  // all links have been selected
159  if (!checkLinkPresent(link)) {
160  continue;
161  }
162 
163  if (mVerbosity) {
164  std::cout << "Finding sync pattern for link " << link << std::endl;
165  std::cout << "Num packets : " << mPacketsPerLink[link] << std::endl;
166  }
167 
168  GBTFrame gFrame;
169  uint32_t packetID{ 0 };
170 
171  // loop over the packets for each link and process them
172  for (auto packet : mPacketDescriptorMaps[link]) {
173  gFrame.setPacketID(packetID);
174 
175  file.seekg(packet.getPayloadOffset(), file.beg);
176 
177  // read in the data frame by frame, extract the 5-bit halfwords for
178  // the two data streams and store them in the corresponding half-word
179  // vectors
180  for (int frames = 0; frames < packet.getPayloadSize() / 16; frames++) {
181  file >> gFrame;
182  // extract the half words from the 4 32-bit words
183  gFrame.getFrameHalfWords();
184  gFrame.updateSyncCheck(mSyncPositions[link]);
185  };
186 
187  packetID++;
188 
189  // TODO: In future there might be more then one sync in the stream
190  // this should be takein into account
191  if (syncFoundForLink(link)) {
192  if (CHECK_BIT(mDebugLevel, 0)) {
193  std::cout << "Sync positions for link " << link << '\n';
194  const auto& syncs = mSyncPositions[link];
195  for (int i = 0; i < syncs.size(); ++i) {
196  std::cout << i << " : " << syncs[i];
197  }
198  std::cout << '\n';
199  }
200  break;
201  }
202  }
203  }

206 int RawReaderCRU::processPacket(GBTFrame& gFrame, uint32_t startPos, uint32_t size, ADCRawData& rawData)
207 {
208  // open the data file
209  std::ifstream file;
210  file.open(mInputFileName, std::ifstream::binary);
211  if (!file.good())
212  throw std::runtime_error("Unable to open or access file " + mInputFileName);
213 
214  // jump to the start position of the packet
215  file.seekg(startPos, file.beg);
216 
217  // read in the data frame by frame, extract the 5-bit halfwords for
218  // the two data streams and store them in the corresponding half-word
219  // vectors
220  for (int frames = 0; frames < size / 16; frames++) {
221  file >> gFrame;
222  // backup the halfword of the frame before calculating the
223  // new halfwords. The previous half words might be needed
224  // to decode the ADC values.
225  gFrame.storePrevFrame();
226  // extract the half words from the 4 32-bit words
227  gFrame.getFrameHalfWords();
228 
229  // debug output
230  if (CHECK_BIT(mDebugLevel, 1)) {
231  std::cout << gFrame;
232  }
233 
234  gFrame.getAdcValues(rawData);
235  gFrame.updateSyncCheck(CHECK_BIT(mDebugLevel, 0));
236  if (!(rawData.getNumTimebins() % 16) && (rawData.getNumTimebins() >= mNumTimeBins * 16))
237  return 1;
238  };
239  return 0;
240 }

242 int RawReaderCRU::processData()
243 {
244  GBTFrame gFrame;
245  //gFrame.setSyncPositions(mSyncPositions[mLink]);
246 
247  uint32_t packetID = 0;
248  ADCRawData rawData;
249 
250  if (mVerbosity) {
251  std::cout << "Processing data for link " << mLink << std::endl;
252  std::cout << "Num packets : " << mPacketsPerLink[mLink] << std::endl;
253  }
254 
255  // ===| mapping to be updated |===============================================
256  CRU cru; // assuming each decoder only hast once CRU
257  const int link = mLink;
258 
259  const auto& mapper = Mapper::instance();
260 
261  // loop over the packets for each link and process them
262  for (auto packet : mPacketDescriptorMaps[link]) {
263  cru = packet.getCRUID();
264  // std::cout << "Packet : " << packetID << std::endl;
265  gFrame.setPacketID(packetID);
266  int retCode = processPacket(gFrame, packet.getPayloadOffset(), packet.getPayloadSize(), rawData);
267  if (retCode)
268  break;
269  packetID++;
270 
271  //if (rawData.getNumTimebins() >= mNumTimeBins * 16)
272  //break;
273  };
274 
275  // ===| fill ADC data to the output structure |===
276  if (mFillADCdataMap) {
277  const int fecLinkOffsetCRU = (mapper.getPartitionInfo(CRU(cru).partition()).getNumberOfFECs() + 1) / 2;
278  const int fecInPartition = (mLink % 12) + (mLink > 11) * fecLinkOffsetCRU;
279  const int regionIter = cru % 2;
280 
281  const int sampaMapping[10] = { 0, 0, 1, 1, 2, 3, 3, 4, 4, 2 };
282  const int channelOffset[10] = { 0, 16, 0, 16, 0, 0, 16, 0, 16, 16 };
283 
284  for (int istreamm = 0; istreamm < 5; ++istreamm) {
285  const int partitionStream = istreamm + regionIter * 5;
286  const int sampa = sampaMapping[partitionStream];
287 
288  const auto& dataVector = rawData.getDataVector(istreamm);
289 
290  // loop over all data. Each stream has 16 ADC values for each sampa channel times nTimeBins
291  for (int idata = 0; idata < dataVector.size(); ++idata) {
292  const int ichannel = idata % 16;
293  const int sampaChannel = ichannel + channelOffset[partitionStream];
294  const auto& padPos = mapper.padPosRegion(cru.region(), fecInPartition, sampa, sampaChannel);
295  mADCdata[padPos].emplace_back(dataVector[idata]);
296  }
297  }
298  }
299 
300  // std::cout << "Output Data" << std::endl;
301 
303  if (mDumpTextFiles) {
304  std::ofstream file;
305  std::string fileName;
306  for (int s = 0; s < 5; s++) {
307  if (mStream == 0x0 or ((mStream >> s) & 0x1) == 0x1) {
308  if (gFrame.mSyncFound(s) == false)
309  std::cout << "No sync found" << std::endl;
310  // debug output
311  rawData.setOutputStream(s);
312  rawData.setNumTimebins(mNumTimeBins);
313  if (CHECK_BIT(mDebugLevel, 2)) {
314  std::cout << rawData << std::endl;
315  };
316  // write the data to file
317  //fileName = "ADC_" + std::to_string(mLink) + "_" + std::to_string(s);
318  fileName = mOutputFilePrefix + "_ADC_" + std::to_string(mLink) + "_" + std::to_string(s) + ".txt";
319  file.open(fileName, std::ofstream::out);
320  file << rawData;
321  file.close();
322  }
323  }
324  }
325  return 0;
326 }

328 void RawReaderCRU::processLinks(const uint32_t linkMask)
329 {
330  try {
331  // read the input data from file, create the packet descriptor map
332  // and the mLinkPresent map.
333  scanFile();
334 
335  // loop over the mNumberLinks potential links in the data
336  // only if data from the link is present and selected
337  // for decoding it will be decoded.
338  for (int lnk = 0; lnk < mNumberLinks; lnk++) {
339  // all links have been selected
340  if (linkMask == 0 && checkLinkPresent(lnk) == true) {
341  // set the active link variable and process the data
342  setLink(lnk);
343  processData();
344  } else if (((linkMask >> lnk) & 0x1) == 0x1 && checkLinkPresent(lnk) == true) {
345  // set the active link variable and process the data
346  setLink(lnk);
347  processData();
348  };
349  };
350 
351  } catch (const RawReaderCRU::Error& e) {
352  std::cout << e.what() << std::endl;
353  exit(10);
354  } catch (const std::exception& e) {
355  std::cerr << "ERROR: " << e.what() << std::endl;
356  exit(100);
357  } catch (...) {
358  std::cerr << "ERROR: Unknown error" << std::endl;
359  exit(1000);
360  }
361 }
