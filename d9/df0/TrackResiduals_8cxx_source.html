69 void TrackResiduals::initBinning()
70 {
71  // initialize binning structures
72  //
73  // X binning
74  if (mNXBins > 0 && mNXBins < param::NPadRows) {
75  // uniform binning in X
76  mDXI = mNXBins / (param::MaxX - param::MinX[0]);
77  mDX = 1.0f / mDXI;
78  mUniformBins[VoxX] = true;
79  } else {
80  // binning per pad row
81  mNXBins = param::NPadRows;
82  mUniformBins[VoxX] = false;
83  mDX = param::RowDX[0];
84  mDXI = 1.f / mDX; // should not be used
85  }
86  //
87  // Y/X binning
88  mMaxY2X.resize(mNXBins);
89  mDY2XI.resize(mNXBins);
90  mDY2X.resize(mNXBins);
91  //
92  for (int ix = 0; ix < mNXBins; ++ix) {
93  float x = getX(ix);
94  mMaxY2X[ix] = tan(.5f * SECPHIWIDTH) - sDeadZone / x;
95  mDY2XI[ix] = mNY2XBins / (2.f * mMaxY2X[ix]);
96  mDY2X[ix] = 1.f / mDY2XI[ix];
97  }
98  mUniformBins[VoxF] = true;
99  //
100  // Z binning
101  mDZI = mNZ2XBins / sMaxZ2X;
102  mDZ = 1.0f / mDZI;
103  mUniformBins[VoxZ] = true;
104  //
105  mNVoxPerSector = mNY2XBins * mNZ2XBins * mNXBins;
106 }

140 int TrackResiduals::getXBin(float x) const
141 {
142  // convert x to bin ID, following pad row widths
143  if (mUniformBins[VoxX]) {
144  int ix = (x - param::MinX[0]) * mDXI;
145  if (ix < 0) {
146  ix = 0;
147  }
148  return (ix < mNXBins) ? ix : mNXBins - 1;
149  } else {
150  int ix;
151  if (x < param::MinX[1] - .5f * param::ROCDX[0]) {
152  // we are in the IROC
153  ix = (x - param::MinX[0] + .5f * param::RowDX[0]) / param::RowDX[0];
154  return (ix < 0) ? 0 : std::min(ix, param::NRowsPerROC[0] - 1);
155  } else if (x > param::MinX[param::NROCTypes - 1] - .5f * param::ROCDX[param::NROCTypes - 2]) {
156  // we are in the last OROC
157  ix = param::NRowsAccumulated[param::NROCTypes - 2] + (x - param::MinX[param::NROCTypes - 1] + .5f * param::RowDX[param::NROCTypes - 1]) / param::RowDX[param::NROCTypes - 1];
158  return (ix < param::NRowsAccumulated[param::NROCTypes - 2]) ? param::NRowsAccumulated[param::NROCTypes - 2] : std::min(ix, param::NPadRows - 1);
159  }
160 #ifdef TPC_RUN2
161  else {
162  // we are in OROC1
163  ix = param::NRowsPerROC[0] + (x - param::MinX[1] + .5f * param::RowDX[1]) / param::RowDX[1];
164  return (ix < param::NRowsPerROC[0]) ? param::NRowsPerROC[0] : std::min(ix, param::NRowsAccumulated[1] - 1);
165  }
166 #else
167  else if (x < param::MinX[2] - .5f * param::ROCDX[1]) {
168  // we are in OROC1
169  ix = param::NRowsPerROC[0] + (x - param::MinX[1] + .5f * param::RowDX[1]) / param::RowDX[1];
170  return (ix < param::NRowsPerROC[0]) ? param::NRowsPerROC[0] : std::min(ix, param::NRowsAccumulated[1] - 1);
171  } else {
172  // we are in OROC2
173  ix = param::NRowsAccumulated[1] + (x - param::MinX[2] + .5f * param::RowDX[2]) / param::RowDX[2];
174  return (ix < param::NRowsAccumulated[1]) ? param::NRowsAccumulated[1] : std::min(ix, param::NRowsAccumulated[2] - 1);
175  }
176 #endif
177  }

180 void TrackResiduals::setKernelType(KernelType kernel, float bwX, float bwP, float bwZ, float scX, float scP, float scZ)
181 {
182  // set kernel type and widths in terms of binning in x, y/x, z/x and define aux variables
183  mKernelType = kernel;
184 
185  mKernelScaleEdge[VoxX] = scX;
186  mKernelScaleEdge[VoxF] = scP;
187  mKernelScaleEdge[VoxZ] = scZ;
188 
189  mKernelWInv[VoxX] = (bwX > 0) ? 1. / bwX : 1.;
190  mKernelWInv[VoxF] = (bwP > 0) ? 1. / bwP : 1.;
191  mKernelWInv[VoxZ] = (bwZ > 0) ? 1. / bwZ : 1.;
192 
193  if (mKernelType == KernelType::Epanechnikov) {
194  // bandwidth 1
195  mStepKern[VoxX] = static_cast<int>(nearbyint(bwX + 0.5));
196  mStepKern[VoxF] = static_cast<int>(nearbyint(bwP + 0.5));
197  mStepKern[VoxZ] = static_cast<int>(nearbyint(bwZ + 0.5));
198  } else if (mKernelType == KernelType::Gaussian) {
199  // look in ~5 sigma
200  mStepKern[VoxX] = static_cast<int>(nearbyint(bwX * 5. + 0.5));
201  mStepKern[VoxF] = static_cast<int>(nearbyint(bwP * 5. + 0.5));
202  mStepKern[VoxZ] = static_cast<int>(nearbyint(bwZ * 5. + 0.5));
203  } else {
204  LOG(error) << "given kernel type is not defined";
205  }
206  for (int i = VoxDim; i--;) {
207  if (mStepKern[i] < 1) {
208  mStepKern[i] = 1;
209  }
210  }

231 void TrackResiduals::processSectorResiduals(int iSec)
232 {
233  if (iSec < 0 || iSec > 35) {
234  LOG(error) << "wrong sector: " << iSec;
235  return;
236  }
237  LOG(info) << "processing sector residuals for sector " << iSec;
238  if (!mIsInitialized) {
239  init();
240  }
241  // open file and retrieve data tree (only local files are supported at the moment)
242  std::string filename = mLocalResFileName + std::to_string(iSec) + ".root";
243  std::unique_ptr<TFile> flin = std::make_unique<TFile>(filename.c_str());
244  if (!flin || flin->IsZombie()) {
245  LOG(error) << "failed to open " << filename.c_str();
246  return;
247  }
248  std::string treename = "ts" + std::to_string(iSec);
249  std::unique_ptr<TTree> tree((TTree*)flin->Get(treename.c_str()));
250  if (!tree) {
251  LOG(error) << "did not find the data tree " << treename.c_str();
252  return;
253  }
254  AliTPCDcalibRes::dts_t trkRes;
255  auto* pTrkRes = &trkRes;
256  tree->SetBranchAddress("dts", &pTrkRes);
257  auto nPoints = tree->GetEntries();
258  if (!nPoints) {
259  LOG(warning) << "no entries found for sector " << iSec;
260  flin->Close();
261  return;
262  }
263  if (nPoints > mMaxPointsPerSector) {
264  nPoints = mMaxPointsPerSector;
265  }
266  // initialize container holding results
267  initResultsContainer(iSec);
268 
269  LOG(info) << "extracted " << nPoints << " of unbinned data";
270 
271  std::vector<bres_t>& secData = mVoxelResults[iSec];
272 
273  unsigned int nAccepted = 0;
274 
275  std::vector<float> dyData(nPoints);
276  std::vector<float> dzData(nPoints);
277  std::vector<float> tgSlpData(nPoints);
278  std::vector<unsigned short> binData(nPoints);
279 
280  if (mPrintMem) {
281  printMem();
282  }
283 
284  // read input data into internal vectors
285  for (int i = 0; i < nPoints; ++i) {
286  tree->GetEntry(i);
287  if (fabs(trkRes.tgSlp) >= mMaxTgSlp) {
288  continue;
289  }
290  dyData[nAccepted] = trkRes.dy;
291  dzData[nAccepted] = trkRes.dz;
292  tgSlpData[nAccepted] = trkRes.tgSlp;
293  binData[nAccepted] = getGlbVoxBin(trkRes.bvox[VoxX], trkRes.bvox[VoxF], trkRes.bvox[VoxZ]);
294  nAccepted++;
295  }
296 
297  tree.release();
298  flin->Close();
299 
300  if (mPrintMem) {
301  printMem();
302  }
303 
304  LOG(info) << "Done reading input data (accepted " << nAccepted << " points)";
305 
306  std::vector<size_t> binIndices(nAccepted);
307 
308  dyData.resize(nAccepted);
309  dzData.resize(nAccepted);
310  tgSlpData.resize(nAccepted);
311  binData.resize(nAccepted);
312 
313  // convert to short and back to float to be compatible with AliRoot version
314  std::vector<short> dyDataShort(nAccepted);
315  std::vector<short> dzDataShort(nAccepted);
316  std::vector<short> tgSlpDataShort(nAccepted);
317  for (unsigned int i = 0; i < nAccepted; ++i) {
318  dyDataShort[i] = short(dyData[i] * 0x7fff / sMaxResid);
319  dzDataShort[i] = short(dzData[i] * 0x7fff / sMaxResid);
320  tgSlpDataShort[i] = short(tgSlpData[i] * 0x7fff / mMaxTgSlp);
321 
322  dyData[i] = dyDataShort[i] * sMaxResid / 0x7fff;
323  dzData[i] = dzDataShort[i] * sMaxResid / 0x7fff;
324  tgSlpData[i] = tgSlpDataShort[i] * mMaxTgSlp / 0x7fff;
325  }
326 
327  // sort in voxel increasing order
328  o2::mathUtils::mathBase::SortData(binData, binIndices);
329  if (mPrintMem) {
330  printMem();
331  }
332 
333  // vectors holding the data for one voxel at a time
334  std::vector<float> dyVec;
335  std::vector<float> dzVec;
336  std::vector<float> tgVec;
337  // assuming we will always have around 1000 entries per voxel
338  dyVec.reserve(1e3);
339  dzVec.reserve(1e3);
340  tgVec.reserve(1e3);
341  int currVoxBin = -1;
342  unsigned int nPointsInVox = 0;
343  unsigned int nProcessed = 0;
344  while (nProcessed < nAccepted) {
345  // read all points, voxel by voxel
346  int idx = binIndices[nProcessed];
347  if (currVoxBin != binData[idx]) {
348  if (nPointsInVox) {
349  bres_t& resVox = secData[currVoxBin];
350  processVoxelResiduals(dyVec, dzVec, tgVec, resVox);
351  }
352  currVoxBin = binData[idx];
353  nPointsInVox = 0;
354  dyVec.clear();
355  dzVec.clear();
356  tgVec.clear();
357  }
358  dyVec.push_back(dyData[idx]);
359  dzVec.push_back(dzData[idx]);
360  tgVec.push_back(tgSlpData[idx]);
361  ++nPointsInVox;
362  ++nProcessed;
363  }
364  if (nPointsInVox) {
365  // process last voxel
366  bres_t& resVox = secData[currVoxBin];
367  processVoxelResiduals(dyVec, dzVec, tgVec, resVox);
368  }
369  LOG(info) << "extracted residuals for sector " << iSec;
370 
371  int nRowsOK = validateVoxels(iSec);
372  LOG(info) << "number of validated X rows: " << nRowsOK;
373  if (!nRowsOK) {
374  LOG(warning) << "sector " << iSec << ": all X-bins disabled, abandon smoothing";
375  return;
376  } else {
377  smooth(iSec);
378  }
379 
380  // process dispersions
381  dyVec.clear();
382  tgVec.clear();
383  currVoxBin = -1;
384  nProcessed = 0;
385  nPointsInVox = 0;
386  while (nProcessed < nAccepted) {
387  int idx = binIndices[nProcessed];
388  if (currVoxBin != binData[idx]) {
389  if (nPointsInVox) {
390  bres_t& resVox = secData[currVoxBin];
391  if (!getXBinIgnored(iSec, resVox.bvox[VoxX])) {
392  processVoxelDispersions(tgVec, dyVec, resVox);
393  }
394  }
395  currVoxBin = binData[idx];
396  nPointsInVox = 0;
397  dyVec.clear();
398  tgVec.clear();
399  }
400  dyVec.push_back(dyData[idx]);
401  tgVec.push_back(tgSlpData[idx]);
402  ++nPointsInVox;
403  ++nProcessed;
404  }
405  if (nPointsInVox) {
406  // process last voxel
407  bres_t& resVox = secData[currVoxBin];
408  if (!getXBinIgnored(iSec, resVox.bvox[VoxX])) {
409  processVoxelDispersions(tgVec, dyVec, resVox);
410  }
411  }
412  // smooth dispersions
413  for (int ix = 0; ix < mNXBins; ++ix) {
414  if (getXBinIgnored(iSec, ix)) {
415  continue;
416  }
417  for (int iz = 0; iz < mNZ2XBins; ++iz) {
418  for (int ip = 0; ip < mNY2XBins; ++ip) {
419  int voxBin = getGlbVoxBin(ix, ip, iz);
420  bres_t& resVox = secData[voxBin];
421  getSmoothEstimate(iSec, resVox.stat[VoxX], resVox.stat[VoxF], resVox.stat[VoxZ], resVox.DS, 0x1 << VoxV);
422  }
423  }
424  }
425  dumpResults(iSec);
426 }

429 void TrackResiduals::processVoxelResiduals(std::vector<float>& dy, std::vector<float>& dz, std::vector<float>& tg, bres_t& resVox)
430 {
431  size_t nPoints = dy.size();
432  //LOG(debug) << "processing voxel residuals for vox " << getGlbVoxBin(resVox.bvox) << " with " << nPoints << " points";
433  if (nPoints < mMinEntriesPerVoxel) {
434  LOG(info) << "voxel " << getGlbVoxBin(resVox.bvox) << " is skipped due to too few entries (" << nPoints << " < " << mMinEntriesPerVoxel << ")";
435  return;
436  }
437  std::array<float, 7> zResults;
438  resVox.flags = 0;
439  std::vector<size_t> indices(dz.size());
440  if (!o2::mathUtils::mathBase::LTMUnbinned(dz, indices, zResults, mLTMCut)) {
441  LOG(debug) << "failed trimming input array for voxel " << getGlbVoxBin(resVox.bvox);
442  return;
443  }
444  std::array<float, 2> res{ 0.f };
445  std::array<float, 3> err{ 0.f };
446  float sigMAD = fitPoly1Robust(tg, dy, res, err, mLTMCut);
447  if (sigMAD < 0) {
448  LOG(debug) << "failed robust linear fit, sigMAD = " << sigMAD;
449  return;
450  }
451  float corrErr = err[0] * err[2];
452  corrErr = corrErr > 0 ? err[1] / std::sqrt(corrErr) : -999;
453  //
454  resVox.D[ResX] = -res[1];
455  resVox.D[ResY] = res[0];
456  resVox.D[ResZ] = zResults[1];
457  resVox.E[ResX] = std::sqrt(err[2]);
458  resVox.E[ResY] = std::sqrt(err[0]);
459  resVox.E[ResZ] = zResults[4];
460  resVox.EXYCorr = corrErr;
461  resVox.D[ResD] = resVox.dYSigMAD = sigMAD; // later will be overwritten by real dispersion
462  resVox.dZSigLTM = zResults[2];
463  //
464  //
465  // at this point the actual COG for each voxel should be stored in resVox.stat
466 
467  resVox.flags |= DistDone;
468 
469  return;
470 }

472 void TrackResiduals::processVoxelDispersions(std::vector<float>& tg, std::vector<float>& dy, bres_t& resVox)
473 {
474  size_t nPoints = tg.size();
475  LOG(debug) << "processing voxel dispersions for vox " << getGlbVoxBin(resVox.bvox) << " with " << nPoints << " points";
476  if (nPoints < 2) {
477  return;
478  }
479  for (size_t i = nPoints; i--;) {
480  dy[i] -= resVox.DS[ResY] - resVox.DS[ResX] * tg[i];
481  }
482  resVox.D[ResD] = getMAD2Sigma(dy);
483  resVox.E[ResD] = resVox.D[ResD] / sqrt(2.f * nPoints); // a la gaussioan RMS error (very crude)
484  resVox.flags |= DispDone;
485 }

487 //______________________________________________________________________________
488 int TrackResiduals::validateVoxels(int iSec)
489 {
490  // apply voxel validation cuts
491  // return number of good voxels for given sector
492  int cntMasked = 0; // number of voxels masked due to fit error and / or distribution sigmas
493  int cntInvalid = 0; // number of voxels which were invalid before + masked ones
494  mXBinsIgnore[iSec].reset();
495  std::vector<bres_t>& secData = mVoxelResults[iSec];
496 
497  int cntMaskedFit = 0;
498  int cntMaskedSigma = 0;
499 
500  // find bad voxels in sector
501  for (int ix = 0; ix < mNXBins; ++ix) {
502  int cntValid = 0;
503  for (int ip = 0; ip < mNY2XBins; ++ip) {
504  for (int iz = 0; iz < mNZ2XBins; ++iz) {
505  int binGlb = getGlbVoxBin(ix, ip, iz);
506  bres_t& resVox = secData[binGlb];
507  bool voxelOK = (resVox.flags & DistDone) && !(resVox.flags & Masked);
508  if (voxelOK) {
509  // check fit errors
510  if (resVox.E[ResY] * resVox.E[ResY] > mMaxFitErrY2 ||
511  resVox.E[ResX] * resVox.E[ResX] > mMaxFitErrX2 ||
512  fabs(resVox.EXYCorr) > mMaxFitCorrXY) {
513  voxelOK = false;
514  ++cntMaskedFit;
515  }
516  // check raw distribution sigmas
517  if (resVox.dYSigMAD > mMaxSigY ||
518  resVox.dZSigLTM > mMaxSigZ) {
519  voxelOK = false;
520  ++cntMaskedSigma;
521  }
522  if (!voxelOK) {
523  ++cntMasked;
524  }
525  }
526  if (voxelOK) {
527  ++cntValid;
528  } else {
529  ++cntInvalid;
530  resVox.flags |= Masked;
531  }
532  } // loop over Z
533  } // loop over Y/X
534  mValidFracXBins[iSec][ix] = static_cast<float>(cntValid) / (mNY2XBins * mNZ2XBins);
535  LOG(debug) << "sector " << iSec << ": xBin " << ix << " has " << mValidFracXBins[iSec][ix] * 100 << "\% of voxels valid";
536  } // loop over X
537 
538  // mask X-bins which cannot be smoothed
539 
540  short nBadReg = 0; // count bad regions (one or more consecutive bad X-bins)
541  std::array<short, param::NPadRows> badStart; // to store indices to the beginnings of the bad regions
542  std::array<short, param::NPadRows> badEnd; // to store indices to the end of the bad regions
543  bool prevBad = false;
544  float fracBadRows = 0.f;
545  for (int ix = 0; ix < mNXBins; ++ix) {
546  if (mValidFracXBins[iSec][ix] < mMinValidVoxFracDrift) {
547  LOG(debug) << "row " << ix << " is bad";
548  ++fracBadRows;
549  if (prevBad) {
550  badEnd[nBadReg] = ix;
551  } else {
552  badStart[nBadReg] = ix;
553  badEnd[nBadReg] = ix;
554  prevBad = true;
555  }
556  } else {
557  if (prevBad) {
558  ++nBadReg;
559  prevBad = false;
560  }
561  }
562  }
563  if (prevBad) {
564  ++nBadReg;
565  }
566  fracBadRows /= mNXBins;
567  if (fracBadRows > mMaxFracBadRowsPerSector) {
568  LOG(warning) << "sector " << iSec << ": Fraction of bad X-bins: " << fracBadRows << " -> masking whole sector";
569  mXBinsIgnore[iSec].set();
570  } else {
571  for (int iBad = 0; iBad < nBadReg; ++iBad) {
572  LOG(debug) << "masking bad region " << iBad;
573  short badInReg = badEnd[iBad] - badStart[iBad] + 1;
574  short badInNextReg = iBad < (nBadReg - 1) ? badEnd[iBad] - badStart[iBad] + 1 : 0;
575  if (badInReg > mMaxBadXBinsToCover) {
576  // disable too large bad patches
577  for (int i = 0; i < badInReg; ++i) {
578  LOG(debug) << "disabling too large patch in bad region " << iBad << ", badStart(" << badStart[iBad] << "), i(" << i << ")";
579  mXBinsIgnore[iSec].set(badStart[iBad] + i);
580  }
581  }
582  if (badInNextReg > mMaxBadXBinsToCover && (badStart[iBad + 1] - badEnd[iBad] - 1) < mMinGoodXBinsToCover) {
583  // disable too small isolated good patches`
584  for (int i = badEnd[iBad] + 1; i < badStart[iBad + 1]; ++i) {
585  LOG(debug) << "disabling too small good patch before bad region " << iBad + 1 << ", badStart(" << badEnd[iBad] << "), badEnd(" << badStart[iBad + 1] << ")";
586  mXBinsIgnore[iSec].set(i);
587  }
588  }
589  }
590  if (nBadReg) {
591  if (mXBinsIgnore[iSec].test(badStart[0]) && badStart[0] < mMinGoodXBinsToCover) {
592  // 1st good patch is too small
593  for (int i = 0; i < badStart[0]; ++i) {
594  LOG(debug) << "disabling too small first good patch badStart(0), badEnd(" << badStart[0] << ")";
595  mXBinsIgnore[iSec].set(i);
596  }
597  }
598  if (mXBinsIgnore[iSec].test(badStart[nBadReg - 1]) && (mNXBins - badEnd[nBadReg - 1] - 1) < mMinGoodXBinsToCover) {
599  // last good patch is too small
600  for (int i = badEnd[nBadReg - 1] + 1; i < mNXBins; ++i) {
601  LOG(debug) << "disabling too small last good patch badStart(" << badEnd[nBadReg - 1] << "), badEnd(" << mNXBins << ")";
602  mXBinsIgnore[iSec].set(i);
603  }
604  }
605  }
606  }
607  //
608  int nMaskedRows = mXBinsIgnore[iSec].count();
609  LOG(info) << "sector " << iSec << ": voxel stat: masked: " << cntMasked << " invalid: " << cntInvalid - cntMasked;
610  //
611  return mNXBins - nMaskedRows;

619  continue;
620  }
621  for (int ip = 0; ip < mNY2XBins; ++ip) {

627  if (!res) {
628  mNSmoothingFailedBins[iSec]++;
629  } else {
630  resVox.flags |= SmoothDone;
631  }
632  }
633  }
634  }
635  // substract dX contribution to dZ
636  for (int ix = 0; ix < mNXBins; ++ix) {

639  }

654 bool TrackResiduals::getSmoothEstimate(int iSec, float x, float p, float z, std::array<float, ResDim>& res, int whichDim)
655 {
656  // get smooth estimate for distortions for point in sector coordinates
658 
659  std::array<int, VoxDim> minPointsDir{ 0 }; // min number of points per direction
660  const float kTrialStep = 0.5;
661  std::array<bool, ResDim> doDim{ false };
662  for (int i = 0; i < ResDim; ++i) {
663  doDim[i] = (whichDim & (0x1 << i)) > 0;
664  if (doDim[i]) {
665  res[i] = 0.f;
666  }
667  }
668 
669  int matSize = sSmtLinDim;
670  for (int i = 0; i < VoxDim; ++i) {
671  minPointsDir[i] = 3; // for pol1 smoothing require at least 3 points
672  if (mSmoothPol2[i]) {
673  ++minPointsDir[i];
674  ++matSize;
675  }
676  }
677 
678  int ix0, ip0, iz0;
679  findVoxel(x, p, iSec < SECTORSPERSIDE ? z : -z, ix0, ip0, iz0); // find nearest voxel
680  std::vector<bres_t>& secData = mVoxelResults[iSec];
681  int binCenter = getGlbVoxBin(ix0, ip0, iz0); // global bin of nearest voxel
682  bres_t& voxCenter = secData[binCenter]; // nearest voxel
683  LOG(debug) << "getting smooth estimate around voxel " << binCenter;
684 
685  // cache
686  // \todo maybe a 1-D cache would be more efficient?
687  std::array<std::array<double, sMaxSmtDim*(sMaxSmtDim + 1) / 2>, ResDim> cmat;
688  int maxNeighb = 10 * 10 * 10;
689  std::vector<bres_t*> currVox;
690  currVox.reserve(maxNeighb);
691  std::vector<float> currCache;
692  currCache.reserve(maxNeighb * VoxHDim);
693 
694  std::array<int, VoxDim> maxTrials;
695  maxTrials[VoxZ] = mNZ2XBins / 2;
696  maxTrials[VoxF] = mNY2XBins / 2;
697  maxTrials[VoxX] = mMaxBadXBinsToCover * 2;
698 
699  std::array<int, VoxDim> trial{ 0 };
700 
701  while (true) {
702  std::fill(mLastSmoothingRes.begin(), mLastSmoothingRes.end(), 0);
703  memset(&cmat[0][0], 0, sizeof(cmat));
704 
705  int nbOK = 0; // accounted neighbours
706 
707  float stepX = mStepKern[VoxX] * (1. + kTrialStep * trial[VoxX]);
708  float stepF = mStepKern[VoxF] * (1. + kTrialStep * trial[VoxF]);
709  float stepZ = mStepKern[VoxZ] * (1. + kTrialStep * trial[VoxZ]);
710 
711  if (!(voxCenter.flags & DistDone) || (voxCenter.flags & Masked) || getXBinIgnored(iSec, ix0)) {
712  // closest voxel has no data -> increase smoothing step
713  stepX += kTrialStep * mStepKern[VoxX];
714  stepF += kTrialStep * mStepKern[VoxF];
715  stepZ += kTrialStep * mStepKern[VoxZ];
716  }
717 
718  // effective kernel widths accounting for the increased bandwidth at the edges and missing data
719  float kWXI = getDXI(ix0) * mKernelWInv[VoxX] * mStepKern[VoxX] / stepX;
720  float kWFI = getDY2XI(ix0) * mKernelWInv[VoxF] * mStepKern[VoxF] / stepF;
721  float kWZI = getDZ2XI() * mKernelWInv[VoxZ] * mStepKern[VoxZ] / stepZ;
722  int iStepX = static_cast<int>(nearbyint(stepX + 0.5));
723  int iStepF = static_cast<int>(nearbyint(stepF + 0.5));
724  int iStepZ = static_cast<int>(nearbyint(stepZ + 0.5));
725  int ixMin = ix0 - iStepX;
726  int ixMax = ix0 + iStepX;
727  if (ixMin < 0) {
728  ixMin = 0;
729  ixMax = std::min(static_cast<int>(nearbyint(ix0 + stepX * mKernelScaleEdge[VoxX])), mNXBins - 1);
730  kWXI /= mKernelScaleEdge[VoxX];
731  }
732  if (ixMax >= mNXBins) {
733  ixMax = mNXBins - 1;
734  ixMin = std::max(static_cast<int>(nearbyint(ix0 - stepX * mKernelScaleEdge[VoxX])), 0);
735  kWXI /= mKernelScaleEdge[VoxX];
736  }
737 
738  int ipMin = ip0 - iStepF;
739  int ipMax = ip0 + iStepF;
740  if (ipMin < 0) {
741  ipMin = 0;
742  ipMax = std::min(static_cast<int>(nearbyint(ip0 + stepF * mKernelScaleEdge[VoxF])), mNY2XBins - 1);
743  kWFI /= mKernelScaleEdge[VoxF];
744  }
745  if (ipMax >= mNY2XBins) {
746  ipMax = mNY2XBins - 1;
747  ipMin = std::max(static_cast<int>(nearbyint(ip0 - stepF * mKernelScaleEdge[VoxF])), 0);
748  kWFI /= mKernelScaleEdge[VoxF];
749  }
750 
751  int izMin = iz0 - iStepZ;
752  int izMax = iz0 + iStepZ;
753  if (izMin < 0) {
754  izMin = 0;
755  izMax = std::min(static_cast<int>(nearbyint(iz0 + stepZ * mKernelScaleEdge[VoxZ])), mNZ2XBins - 1);
756  kWZI /= mKernelScaleEdge[VoxZ];
757  }
758  if (izMax >= mNZ2XBins) {
759  izMax = mNZ2XBins - 1;
760  izMin = std::max(static_cast<int>(nearbyint(iz0 - stepZ * mKernelScaleEdge[VoxZ])), 0);
761  kWZI /= mKernelScaleEdge[VoxZ];
762  }
763 
764  std::vector<unsigned short> nOccX(ixMax - ixMin + 1, 0);
765  std::vector<unsigned short> nOccF(ipMax - ipMin + 1, 0);
766  std::vector<unsigned short> nOccZ(izMax - izMin + 1, 0);
767 
768  int nbCheck = (ixMax - ixMin + 1) * (ipMax - ipMin + 1) * (izMax - izMin + 1);
769  if (nbCheck >= maxNeighb) {
770  maxNeighb = nbCheck + 100;
771  currCache.reserve(maxNeighb * VoxHDim);
772  currVox.reserve(maxNeighb);
773  }
774  std::array<double, 3> u2Vec;
775 
776  // first loop, check presence of enough points
777  for (int ix = ixMin; ix <= ixMax; ++ix) {
778  for (int ip = ipMin; ip <= ipMax; ++ip) {
779  for (int iz = izMin; iz <= izMax; ++iz) {
780  int binNb = getGlbVoxBin(ix, ip, iz);
781  bres_t& voxNb = secData[binNb];
782  if (!(voxNb.flags & DistDone) ||
783  (voxNb.flags & Masked) ||
784  getXBinIgnored(iSec, ix)) {
785  // skip voxels w/o data
786  continue;
787  }
788  // estimate weighted distance
789  float dx = voxNb.stat[VoxX] - x;
790  float df = voxNb.stat[VoxF] - p;
791  float dz = voxNb.stat[VoxZ] - z;
792  float dxw = dx * kWXI;
793  float dfw = df * kWFI;
794  float dzw = dz * kWZI;
795  u2Vec[0] = dxw * dxw;
796  u2Vec[1] = dfw * dfw;
797  u2Vec[2] = dzw * dzw;
798  double kernelWeight = getKernelWeight(u2Vec);
799  if (kernelWeight < 1e-6) {
800  continue;
801  }
802  // new point is validated
803  ++nOccX[ix - ixMin];
804  ++nOccF[ip - ipMin];
805  ++nOccZ[iz - izMin];
806  currVox[nbOK] = &voxNb;
807  currCache[nbOK * VoxHDim + VoxX] = dx;
808  currCache[nbOK * VoxHDim + VoxF] = df;
809  currCache[nbOK * VoxHDim + VoxZ] = dz;
810  currCache[nbOK * VoxHDim + VoxV] = kernelWeight;
811  ++nbOK;
812  }
813  }
814  }
815 
816  // check if we have enough points in every dimension
817  std::array<int, VoxDim> nPoints{ 0 };
818  for (int i = ixMax - ixMin + 1; i--;) {
819  if (nOccX[i]) {
820  ++nPoints[VoxX];
821  }
822  }
823  for (int i = ipMax - ipMin + 1; i--;) {
824  if (nOccF[i]) {
825  ++nPoints[VoxF];
826  }
827  }
828  for (int i = izMax - izMin + 1; i--;) {
829  if (nOccZ[i]) {
830  ++nPoints[VoxZ];
831  }
832  }
833  bool enoughPoints = true;
834  std::array<bool, VoxDim> incrDone{ false };
835  for (int i = 0; i < VoxDim; ++i) {
836  if (nPoints[i] < minPointsDir[i]) {
837  // need to extend smoothing neighbourhood
838  enoughPoints = false;
839  if (trial[i] < maxTrials[i] && !incrDone[i]) {
840  // try to increment only missing direction
841  ++trial[i];
842  incrDone[i] = true;
843  } else if (trial[i] == maxTrials[i]) {
844  // cannot increment missing direction, try others
845  for (int j = VoxDim; j--;) {
846  if (i != j && trial[j] < maxTrials[j] && !incrDone[j]) {
847  ++trial[j];
848  incrDone[j] = true;
849  }
850  }
851  }
852  }
853  }
854 
855  if (!enoughPoints) {
856  if (!(incrDone[VoxX] || incrDone[VoxF] || incrDone[VoxZ])) {
857  LOG(error) << "trial limit reached, skipping this voxel";
858  return false;
859  }
860  LOG(debug) << "sector " << iSec << ": increasing filter bandwidth around voxel " << binCenter;
861  //printf("Sector:%2d x=%.2f y/x=%.2f z/x=%.2f (iX: %d iY2X:%d iZ2X:%d)\n", iSec, x, p, z, ix0, ip0, iz0);
862  //printf("not enough neighbours (need min %d) %d %d %d (tot: %d) | Steps: %.1f %.1f %.1f\n", 2, nPoints[VoxX], nPoints[VoxF], nPoints[VoxZ], nbOK, stepX, stepF, stepZ);
863  //printf("trying to increase filter bandwidth (trialXFZ: %d %d %d)\n", trial[VoxX], trial[VoxF], trial[VoxZ]);
864  continue;
865  }
866 
867  // now fill matrices and solve
868  for (int iNb = 0; iNb < nbOK; ++iNb) {
869  double wiCache = currCache[iNb * VoxHDim + VoxV];
870  double dxi = currCache[iNb * VoxHDim + VoxX];
871  double dfi = currCache[iNb * VoxHDim + VoxF];
872  double dzi = currCache[iNb * VoxHDim + VoxZ];
873  double dxi2 = dxi * dxi;
874  double dfi2 = dfi * dfi;
875  double dzi2 = dzi * dzi;
876  const bres_t* voxNb = currVox[iNb];
877  for (int iDim = 0; iDim < ResDim; ++iDim) {
878  if (!doDim[iDim]) {
879  continue;
880  }
881  double vi = voxNb->D[iDim];
882  double wi = wiCache;
883  if (mUseErrInSmoothing && fabs(voxNb->E[iDim]) > 1e-6) {
884  // account for point error apart from kernel value
885  wi /= (voxNb->E[iDim] * voxNb->E[iDim]);
886  }
887  std::array<double, sMaxSmtDim*(sMaxSmtDim + 1) / 2>& cmatD = cmat[iDim];
888  double* rhsD = &mLastSmoothingRes[iDim * sMaxSmtDim];
889  unsigned short iMat = 0;
890  unsigned short iRhs = 0;
891  // linear part
892  cmatD[iMat++] += wi;
893  rhsD[iRhs++] += wi * vi;
894  //
895  cmatD[iMat++] += wi * dxi;
896  cmatD[iMat++] += wi * dxi2;
897  rhsD[iRhs++] += wi * dxi * vi;
898  //
899  cmatD[iMat++] += wi * dfi;
900  cmatD[iMat++] += wi * dxi * dfi;
901  cmatD[iMat++] += wi * dfi2;
902  rhsD[iRhs++] += wi * dfi * vi;
903  //
904  cmatD[iMat++] += wi * dzi;
905  cmatD[iMat++] += wi * dxi * dzi;
906  cmatD[iMat++] += wi * dfi * dzi;
907  cmatD[iMat++] += wi * dzi2;
908  rhsD[iRhs++] += wi * dzi * vi;
909  //
910  // check if quadratic part is needed
911  if (mSmoothPol2[VoxX]) {
912  cmatD[iMat++] += wi * dxi2;
913  cmatD[iMat++] += wi * dxi * dxi2;
914  cmatD[iMat++] += wi * dfi * dxi2;
915  cmatD[iMat++] += wi * dzi * dxi2;
916  cmatD[iMat++] += wi * dxi2 * dxi2;
917  rhsD[iRhs++] += wi * dxi2 * vi;
918  }
919  if (mSmoothPol2[VoxF]) {
920  cmatD[iMat++] += wi * dfi2;
921  cmatD[iMat++] += wi * dxi * dfi2;
922  cmatD[iMat++] += wi * dfi * dfi2;
923  cmatD[iMat++] += wi * dzi * dfi2;
924  cmatD[iMat++] += wi * dxi2 * dfi2;
925  cmatD[iMat++] += wi * dfi2 * dfi2;
926  rhsD[iRhs++] += wi * dfi2 * vi;
927  }
928  if (mSmoothPol2[VoxZ]) {
929  cmatD[iMat++] += wi * dzi2;
930  cmatD[iMat++] += wi * dxi * dzi2;
931  cmatD[iMat++] += wi * dfi * dzi2;
932  cmatD[iMat++] += wi * dzi * dzi2;
933  cmatD[iMat++] += wi * dxi2 * dzi2;
934  cmatD[iMat++] += wi * dfi2 * dzi2;
935  cmatD[iMat++] += wi * dzi2 * dzi2;
936  rhsD[iRhs++] += wi * dzi2 * vi;
937  }
938  }
939  }
940 
941  bool fitRes = true;
942 
943  // solve system of linear equations
944 
945  TMatrixDSym matrix(matSize);
946  TDecompChol chol(matSize);
947  TVectorD rhsVec(matSize);
948  for (int iDim = 0; iDim < ResDim; ++iDim) {
949  if (!doDim[iDim]) {
950  continue;
951  }
952  matrix.Zero(); // reset matrix
953  std::array<double, sMaxSmtDim*(sMaxSmtDim + 1) / 2>& cmatD = cmat[iDim];
954  double* rhsD = &mLastSmoothingRes[iDim * sMaxSmtDim];
955  short iMat = -1;
956  short iRhs = -1;
957  short row = -1;
958 
959  // with the studid implementation of TMatrixDSym we need to set all elements of the matrix explicitly (or maybe only upper triangle?)
960  matrix(++row, 0) = cmatD[++iMat];
961  matrix(++row, 0) = cmatD[++iMat];
962  matrix(row, 1) = cmatD[++iMat];
963  matrix(0, row) = matrix(row, 0);
964  matrix(++row, 0) = cmatD[++iMat];
965  matrix(row, 1) = cmatD[++iMat];
966  matrix(row, 2) = cmatD[++iMat];
967  matrix(0, row) = matrix(row, 0);
968  matrix(1, row) = matrix(row, 1);
969  matrix(++row, 0) = cmatD[++iMat];
970  matrix(row, 1) = cmatD[++iMat];
971  matrix(row, 2) = cmatD[++iMat];
972  matrix(row, 3) = cmatD[++iMat];
973  matrix(0, row) = matrix(row, 0);
974  matrix(1, row) = matrix(row, 1);
975  matrix(2, row) = matrix(row, 2);
976  // add pol2 elements if needed
977  if (mSmoothPol2[VoxX]) {
978  const unsigned int colLim = (++row) + 1;
979  for (int iCol = 0; iCol < colLim; ++iCol) {
980  matrix(row, iCol) = cmatD[++iMat];
981  matrix(iCol, row) = matrix(row, iCol);
982  }
983  }
984  if (mSmoothPol2[VoxF]) {
985  const unsigned int colLim = (++row) + 1;
986  for (int iCol = 0; iCol < colLim; ++iCol) {
987  matrix(row, iCol) = cmatD[++iMat];
988  matrix(iCol, row) = matrix(row, iCol);
989  }
990  }
991  if (mSmoothPol2[VoxZ]) {
992  const unsigned int colLim = (++row) + 1;
993  for (int iCol = 0; iCol < colLim; ++iCol) {
994  matrix(row, iCol) = cmatD[++iMat];
995  matrix(iCol, row) = matrix(row, iCol);
996  }
997  }
998  rhsVec.SetElements(rhsD);
999  chol.SetMatrix(matrix);
1000  chol.Decompose();
1001  fitRes = chol.Solve(rhsVec);
1002  if (!fitRes) {
1003  for (int i = VoxDim; i--;) {
1004  trial[i]++;
1005  }
1006  LOG(error) << "solution for smoothing failed, trying to increase filter bandwidth";
1007  continue;
1008  }
1009  res[iDim] = rhsVec[0];
1010  }
1011 
1012  break;
1013  }
1014 
1015  return true;
1016 }

1018 double TrackResiduals::getKernelWeight(std::array<double, 3> u2vec) const
1019 {
1020  double w = 1.;
1021  if (mKernelType == KernelType::Epanechnikov) {
1022  for (size_t i = u2vec.size(); i--;) {
1023  if (u2vec[i] > 1) {
1024  return 0.;
1025  }
1026  w *= 3. / 4. * (1. - u2vec[i]);
1027  }
1028  } else if (mKernelType == KernelType::Gaussian) {
1029  double u2 = 0.;
1030  for (size_t i = u2vec.size(); i--;) {
1031  u2 += u2vec[i];
1032  }
1033  w = u2 < mMaxGaussStdDev * mMaxGaussStdDev * u2vec.size() ? std::exp(-u2) / std::sqrt(2. * M_PI) : 0;
1034  }
1035  return w;
1036 }

1044 float TrackResiduals::fitPoly1Robust(std::vector<float>& x, std::vector<float>& y, std::array<float, 2>& res, std::array<float, 3>& err, float cutLTM) const
1045 {
1046  // robust pol1 fit, modifies input arrays order
1047  if (x.size() != y.size()) {
1048  LOG(error) << "x and y must not have different sizes for fitPoly1Robust (" << x.size() << " != " << y.size() << ")";
1049  }
1050  size_t nPoints = x.size();
1051  res[0] = res[1] = 0.f;
1052  if (nPoints < 2) {
1053  return -1;
1054  }
1055  std::array<float, 7> yResults;
1056  std::vector<size_t> indY(nPoints);
1057  if (!o2::mathUtils::mathBase::LTMUnbinned(y, indY, yResults, cutLTM)) {
1058  return -1;
1059  }
1060  // rearrange used events in increasing order
1061  o2::mathUtils::mathBase::Reorder(y, indY);
1062  o2::mathUtils::mathBase::Reorder(x, indY);
1063  //

1097 void TrackResiduals::medFit(int nPoints, int offset, const std::vector<float>& x, const std::vector<float>& y, float& a, float& b, std::array<float, 3>& err) const
1098 {
1099  // fitting a straight line y(x|a, b) = a + b * x
1100  // to given x and y data minimizing the absolute deviation
1101  float aa, bb, chi2 = 0.f;
1102  if (nPoints < 2) {
1103  a = b = 0.f;
1104  err[0] = err[1] = err[2] = 999.f;
1105  return;
1106  }
1107  // do least squares minimization as first guess
1108  float sx = 0.f, sxx = 0.f, sy = 0.f, sxy = 0.f;
1109  for (int j = nPoints + offset; j-- > offset;) { // same order as in AliRoot version such that resulting sums are identical
1110  sx += x[j];
1111  sxx += x[j] * x[j];
1112  sy += y[j];
1113  sxy += x[j] * y[j];
1114  }
1115  float del = nPoints * sxx - sx * sx;
1116  float delI = 1. / del;
1117  aa = (sxx * sy - sx * sxy) * delI;
1118  bb = (nPoints * sxy - sx * sy) * delI;
1119  err[0] = sxx * delI;
1120  err[1] = sx * delI;
1121  err[2] = nPoints * delI;
1122 
1123  for (int j = nPoints + offset; j-- > offset;) {
1124  float tmp = y[j] - (aa + bb * x[j]);
1125  chi2 += tmp * tmp;
1126  }
1127  float sigb = std::sqrt(chi2 * delI); // expected sigma for b
1128  float b1 = bb;
1129  float f1 = roFunc(nPoints, offset, x, y, b1, aa);
1130  if (sigb > 0) {
1131  float b2 = bb + std::copysign(3.f * sigb, f1);
1132  float f2 = roFunc(nPoints, offset, x, y, b2, aa);
1133  if (fabs(f1 - f2) < sFloatEps) {
1134  a = aa;
1135  b = bb;
1136  return;
1137  }
1138  while (f1 * f2 > 0.f) {
1139  bb = b2 + 1.6f * (b2 - b1);
1140  b1 = b2;
1141  f1 = f2;
1142  b2 = bb;
1143  f2 = roFunc(nPoints, offset, x, y, b2, aa);
1144  }
1145  sigb = .01f * sigb;
1146  while (fabs(b2 - b1) > sigb) {
1147  bb = b1 + .5f * (b2 - b1);
1148  if (bb == b1 || bb == b2) {
1149  break;
1150  }
1151  float f = roFunc(nPoints, offset, x, y, bb, aa);
1152  if (f * f1 >= .0f) {
1153  f1 = f;
1154  b1 = bb;
1155  } else {
1156  f2 = f;
1157  b2 = bb;
1158  }
1159  }
1160  }
1161  a = aa;
1162  b = bb;
1163 }

1165 float TrackResiduals::roFunc(int nPoints, int offset, const std::vector<float>& x, const std::vector<float>& y, float b, float& aa) const
1166 {
1167  // calculate sum(x_i * sgn(y_i - a - b * x_i)) for given b
1168  // see numberical recipies paragraph 15.7.3
1169  std::vector<float> vecTmp(nPoints);
1170  float sum = 0.f;
1171  for (int j = nPoints; j-- > 0;) {
1172  vecTmp[j] = y[j + offset] - b * x[j + offset];
1173  }
1174  int nPointsHalf = nPoints / 2;
1175  if (nPoints < 20) { // it is faster to do insertion sort
1176  for (int i = 1; i < nPoints; i++) {
1177  float v = vecTmp[i];
1178  int j;
1179  for (j = i; j--;) {
1180  if (vecTmp[j] > v) {
1181  vecTmp[j + 1] = vecTmp[j];
1182  } else {
1183  break;
1184  }
1185  vecTmp[j + 1] = v;
1186  }
1187  }
1188  aa = (nPoints & 0x1) ? vecTmp[nPointsHalf] : .5f * (vecTmp[nPointsHalf - 1] + vecTmp[nPointsHalf]);
1189  } else {
1190  std::vector<float>::iterator nth = vecTmp.begin() + vecTmp.size() / 2;
1191  if (nPoints & 0x1) {
1192  std::nth_element(vecTmp.begin(), nth, vecTmp.end());
1193  aa = *nth;
1194  } else {
1195  std::nth_element(vecTmp.begin(), nth - 1, vecTmp.end());
1196  std::nth_element(nth, nth, vecTmp.end());
1197  aa = 0.5 * (*(nth - 1) + *(nth));
1198  }
1199  /*
1200  aa = (nPoints & 0x1) ? selectKthMin(nPointsHalf, vecTmp) : .5f * (selectKthMin(nPointsHalf - 1, vecTmp) + selectKthMin(nPointsHalf, vecTmp));
1201  */
1202  }
1203  for (int j = nPoints; j-- > 0;) {
1204  float d = y[j + offset] - (b * x[j + offset] + aa);
1205  if (y[j + offset] != 0.f) {
1206  d /= fabs(y[j + offset]);
1207  }
1208  if (fabs(d) > sFloatEps) {
1209  sum += (d >= 0.f ? x[j + offset] : -x[j + offset]);
1210  }
1211  }
1212  return sum;
1213 }

1216 float TrackResiduals::selectKthMin(const int k, std::vector<float>& data)
1217 {
1218  // Returns the k th smallest value in the vector. The input vector will be rearranged
1219  // to have this value in location data[k] , with all smaller elements moved before it
1220  // (in arbitrary order) and all larger elements after (also in arbitrary order).
1221  // From Numerical Recipes in C++ (paragraph 8.5)
1222 
1223  int i, ir, j, l, mid, n = data.size();
1224  float a; // partitioning element
1225  l = 0; // left hand side of active partition
1226  ir = n - 1; // right hand side of active partition
1227 
1228  while (true) {
1229  if (ir <= l + 1) { // active partition with 1 or 2 elements
1230  if (ir == l + 1 && data[ir] < data[l]) { // case of 2 elements
1231  std::swap(data[l], data[ir]);
1232  }
1233  return data[k];
1234  } else {
1235  mid = (l + ir) >> 1;
1236  std::swap(data[mid], data[l + 1]);
1237  if (data[l] > data[ir]) {
1238  std::swap(data[l], data[ir]);
1239  }
1240  if (data[l + 1] > data[ir]) {
1241  std::swap(data[l + 1], data[ir]);
1242  }
1243  if (data[l] > data[l + 1]) {
1244  std::swap(data[l], data[l + 1]);
1245  }
1246  i = l + 1; // initialize pointers for partitioning
1247  j = ir;
1248  a = data[l + 1];
1249  while (true) {
1250  // innermost loop used for partitioning
1251  do {
1252  i++;
1253  } while (data[i] < a);
1254  do {
1255  j--;
1256  } while (data[j] > a);
1257  if (j < i) {
1258  // pointers crossed -> partitioning complete
1259  break;
1260  }
1261  std::swap(data[i], data[j]);
1262  }
1263  data[l + 1] = data[j];
1264  data[j] = a;
1265  // keep the partition which contains kth element active
1266  if (j >= k) {
1267  ir = j - 1;
1268  }
1269  if (j <= k) {
1270  l = i;
1271  }
1272  }
1273  }
1274 }

1277 float TrackResiduals::getMAD2Sigma(std::vector<float> data) const
1278 {
1279  // Sigma calculated from median absolute deviations
1280  // see: https://en.wikipedia.org/wiki/Median_absolute_deviation
1281  // the data is passed by value (copied!), such that the original vector
1282  // is not rearranged
1283 
1284  int nPoints = data.size();
1285  if (nPoints < 2) {
1286  return 0;
1287  }
1288 
1289  // calculate median of the input data
1290  float medianOfData;
1291  std::vector<float>::iterator nth = data.begin() + data.size() / 2;
1292  if (nPoints & 0x1) {
1293  std::nth_element(data.begin(), nth, data.end());
1294  medianOfData = *nth;
1295  } else {
1296  std::nth_element(data.begin(), nth - 1, data.end());
1297  std::nth_element(nth, nth, data.end());
1298  medianOfData = .5f * (*(nth - 1) + (*nth));
1299  }
1300 
1301  // fill vector with absolute deviations to median
1302  for (auto& entry : data) {
1303  entry = fabs(entry - medianOfData);
1304  }
1305 
1306  // calculate median of abs deviations
1307  float medianOfAbsDeviations;
1308  if (nPoints & 0x1) {
1309  std::nth_element(data.begin(), nth, data.end());
1310  medianOfAbsDeviations = *nth;
1311  } else {
1312  std::nth_element(data.begin(), nth - 1, data.end());
1313  std::nth_element(nth, nth, data.end());
1314  medianOfAbsDeviations = .5f * (*(nth - 1) + (*nth));
1315  }
1316 
1317  float k = 1.4826f; // scale factor for normally distributed data
1318  return k * medianOfAbsDeviations;
1319 }
