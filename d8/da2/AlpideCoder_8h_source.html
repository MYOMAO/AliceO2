105  int decodeChip(ChipPixelData& chipData, T& buffer) const
106  {
107  // read record for single non-empty chip, updating on change module and cycle.
108  // return number of records filled (>0), EOFFlag or Error
109  // NOTE: decoder does not clean the chipData buffers, should be done outside
110  //
111  uint8_t dataC = 0, timestamp = 0;
112  uint16_t dataS = 0, region = 0;
113  //
114  int nRightCHits = 0; // counter for the hits in the right column of the current double column
115  std::uint16_t rightColHits[NRows]; // buffer for the accumulation of hits in the right column
116  std::uint16_t colDPrev = 0xffff; // previously processed double column (to dected change of the double column)
117 
118  uint32_t expectInp = ExpectChipHeader | ExpectChipEmpty; // data must always start with chip header or chip empty flag
119 
120  while (buffer.next(dataC)) {
121  //
122  // ---------- chip info ?
123  uint8_t dataCM = dataC & (~MaskChipID);
124  //
125  if ((expectInp & ExpectChipEmpty) && dataCM == CHIPEMPTY) { // chip trailer was expected
126  chipData.setChipID(dataC & MaskChipID); // here we set the chip ID within the module
127  if (!buffer.next(timestamp)) {
128  return unexpectedEOF("CHIP_EMPTY:Timestamp");
129  }
130  expectInp = ExpectChipHeader | ExpectChipEmpty;
131  continue;
132  }
133 
134  if ((expectInp & ExpectChipHeader) && dataCM == CHIPHEADER) { // chip header was expected
135  chipData.setChipID(dataC & MaskChipID); // here we set the chip ID within the module
136  if (!buffer.next(timestamp)) {
137  return unexpectedEOF("CHIP_HEADER");
138  }
139  expectInp = ExpectRegion; // now expect region info
140  continue;
141  }
142 
143  // region info ?
144  if ((expectInp & ExpectRegion) && (dataC & REGION) == REGION) { // chip header was seen, or hit data read
145  region = dataC & MaskRegion;
146  expectInp = ExpectData;
147  continue;
148  }
149 
150  if ((expectInp & ExpectChipTrailer) && dataCM == CHIPTRAILER) { // chip trailer was expected
151  expectInp = ExpectChipHeader | ExpectChipEmpty;
152  chipData.setROFlags(dataC & MaskROFlags);
153  // in case there are entries in the "right" columns buffer, add them to the container
154  if (nRightCHits) {
155  colDPrev++;
156  for (int ihr = 0; ihr < nRightCHits; ihr++) {
157  chipData.getData().emplace_back(rightColHits[ihr], colDPrev);
158  }
159  }
160  break;
161  }
162 
163  // hit info ?
164  if ((expectInp & ExpectData)) {
165  if (isData(dataC)) { // region header was seen, expect data
166  // note that here we are checking on the byte rather than the short, need complete to ushort
167  dataS = dataC << 8;
168  if (!buffer.next(dataC)) {
169  return unexpectedEOF("CHIPDATA");
170  }
171  dataS |= dataC;
172  // we are decoding the pixel addres, if this is a DATALONG, we will fetch the mask later
173  uint16_t dColID = (dataS & MaskEncoder) >> 10;
174  uint16_t pixID = dataS & MaskPixID;
175 
176  // convert data to usual row/pixel format
177  uint16_t row = pixID >> 1;
178  // abs id of left column in double column
179  uint16_t colD = (region * NDColInReg + dColID) << 1; // TODO consider <<4 instead of *NDColInReg?
180 
181  // if we start new double column, transfer the hits accumulated in the right column buffer of prev. double column
182  if (colD != colDPrev) {
183  colDPrev++;
184  for (int ihr = 0; ihr < nRightCHits; ihr++) {
185  chipData.getData().emplace_back(rightColHits[ihr], colDPrev);
186  }
187  colDPrev = colD;
188  nRightCHits = 0; // reset the buffer
189  }
190 
191  bool rightC = (row & 0x1) ? !(pixID & 0x1) : (pixID & 0x1); // true for right column / lalse for left
192 
193  // we want to have hits sorted in column/row, so the hits in right column of given double column
194  // are first collected in the temporary buffer
195  // real columnt id is col = colD + 1;
196  if (rightC) {
197  rightColHits[nRightCHits++] = row; // col = colD+1
198  } else {
199  chipData.getData().emplace_back(row, colD); // col = colD, left column hits are added directly to the container
200  }
201 
202  if ((dataS & (~MaskDColID)) == DATALONG) { // multiple hits ?
203  uint8_t hitsPattern = 0;
204  if (!buffer.next(hitsPattern)) {
205  return unexpectedEOF("CHIP_DATA_LONG:Pattern");
206  }
207  for (int ip = 0; ip < HitMapSize; ip++) {
208  if (hitsPattern & (0x1 << ip)) {
209  uint16_t addr = pixID + ip + 1, rowE = addr >> 1;
210  rightC = ((rowE & 0x1) ? !(addr & 0x1) : (addr & 0x1)); // true for right column / lalse for left
211  // the real columnt is int colE = colD + rightC;
212  if (rightC) { // same as above
213  rightColHits[nRightCHits++] = rowE;
214  } else {
215  chipData.getData().emplace_back(rowE, colD + rightC); // left column hits are added directly to the container
216  }
217  }
218  }
219  }
220  } else {
221  LOG(ERROR) << "Expected DataShort or DataLong mask, got : " << dataS;
222  return Error;
223  }
224  expectInp = ExpectChipTrailer | ExpectData | ExpectRegion;
225  continue; // end of DATA(SHORT or LONG) processing
226  }
227 
228  if (!dataC) {
229  break; // 0 padding reached (end of the cable data)
230  }
231  return unexpectedEOF("Unknown word"); // either error
232  }
233 
234  return chipData.getData().size();
235  }
