142  enum PacketStates : int {
143  PacketDone = 0, // Packet finished
144  TransmissionTimeout = 1, // if timeout of transmission (lanes)
145  PacketOverflow = 2, // if max number of packets reached
146  LaneStartsViolation = 3, // if at least 1 lane had a start violation
147  LameTimeouts = 4, // if at least 1 lane timed out

150  };

152  GBTDataTrailer() : GBTWord(0, GBTFlagDataTrailer << 8) {}
153  GBTDataTrailer(int lanesStop, int lanesTimeOut, int packetState) : GBTWord((lanesStop & LANESMask) | ((lanesTimeOut & LANESMask) << 32),
154  (packetState & 0xff) | (GBTFlagDataTrailer << 8)) {}
