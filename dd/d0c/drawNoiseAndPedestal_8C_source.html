111 void addFECInfo()
112 {
113  using namespace o2::TPC;
114  const int event = gPad->GetEvent();
115  if (event != 51) {
116  return;
117  }
118 
119  int binx, biny;
120  float bincx, bincy;
121  TH1* h = GetBinInfoXY(binx, biny, bincx, bincy);
122  if (!h) {
123  return;
124  }
125 
126  const float binValue = h->GetBinContent(binx, biny);
127  const int row = int(TMath::Floor(bincx));
128  const int cpad = int(TMath::Floor(bincy));
129 
130  const auto& mapper = Mapper::instance();
131 
132  const int roc = h->GetUniqueID();
133  if (roc < 0 || roc >= (int)ROC::MaxROC)
134  return;
135  if (row < 0 || row >= (int)mapper.getNumberOfRowsROC(roc))
136  return;
137  const int nPads = mapper.getNumberOfPadsInRowROC(roc, row);
138  const int pad = cpad + nPads / 2;
139  //printf("row %d, cpad %d, pad %d, nPads %d\n", row, cpad, pad, nPads);
140  if (pad < 0 || pad >= (int)nPads) {
141  return;
142  }
143  const int channel = mapper.getPadNumberInROC(PadROCPos(roc, row, pad));
144 
145  const auto& fecInfo = mapper.getFECInfo(PadROCPos(roc, row, pad));
146 
147  TString title("#splitline{#lower[.1]{#scale[.5]{");
148  title += (roc / 18 % 2 == 0) ? "A" : "C";
149  title += Form("%02d (%02d) row: %02d, pad: %03d, globalpad: %05d (in roc)}}}{#scale[.5]{FEC: %02d, Chip: %02d, Chn: %02d, Value: %.3f}}",
150  roc % 18, roc, row, pad, channel, fecInfo.getIndex(), fecInfo.getSampaChip(), fecInfo.getSampaChannel(), binValue);
151 
152  h->SetTitle(title.Data());
153 }
