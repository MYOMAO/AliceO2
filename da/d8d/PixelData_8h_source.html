133  void maskFiredInSample(const ChipPixelData& sample)
134  {
136  const auto& pixelsS = sample.getData();
137  uint32_t nC = mPixels.size();
138  if (!nC) {
139  return;
140  }
141  uint32_t nS = pixelsS.size();
142  if (!nS) {
143  return;
144  }
145  uint32_t itC = 0, itS = 0;
146  while (itC < nC && itS < nS) {
147  auto& pix0 = mPixels[itC];
148  const auto& pixC = pixelsS[itS];
149  if (pix0 == pixC) { // same
150  pix0.setMask();
151  if (mFirstUnmasked == itC++) { // mFirstUnmasked should flag 1st unmasked pixel entry
152  mFirstUnmasked = itC;
153  }
154  itS++;
155  } else if (pix0 < pixC) {
156  itC++;
157  } else {
158  itS++;
159  }
160  }
161  }
