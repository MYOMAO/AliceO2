95  void getChipInfoSW(int chipSW, ChipInfo& chInfo) const
96  {
97  chInfo.id = chipSW;
98  if (chipSW > NChipsSB[IB] + NChipsSB[MB] - 1) { // OB
99  chipSW -= NChipsSB[IB] + NChipsSB[MB];
100  chInfo.ruType = OB;
101  auto dvRU = std::div(chipSW, NChipsPerStaveSB[OB]);
102  chInfo.ru = NStavesSB[IB] + NStavesSB[MB] + dvRU.quot; // RU ID == stave ID
103  chInfo.chOnRU = getChipOnRUInfo(OB, dvRU.rem);
104  } else if (chipSW > NChipsSB[IB] - 1) { // MB
105  chipSW -= NChipsSB[IB];
106  chInfo.ruType = MB;
107  auto dvRU = std::div(chipSW, NChipsPerStaveSB[MB]);
108  chInfo.ru = NStavesSB[IB] + dvRU.quot; // RU ID == stave ID
109  chInfo.chOnRU = getChipOnRUInfo(MB, dvRU.rem);
110  } else { // IB
111  chInfo.ruType = IB;
112  auto dvRU = std::div(chipSW, NChipsPerStaveSB[IB]);
113  chInfo.ru = dvRU.quot; // RU ID == stave ID = module ID
114  chInfo.chOnRU = getChipOnRUInfo(IB, dvRU.rem);
115  }
116  }
