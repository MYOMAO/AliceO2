435 inline void RawReaderCRU::GBTFrame::updateSyncCheck(SyncArray& syncArray)
436 {
437  for (int s = 0; s < 5; s++)
438  for (int h = 0; h < 4; h++) {
439  // shift in a 1 if the halfword is 0x15
440  if (mFrameHalfWords[s][h] == 0x15)
441  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1) | 1;
442  // shift in a 0 if the halfword is 0xA
443  else if (mFrameHalfWords[s][h] == 0xA)
444  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1);
445  // otherwise reset the register to 0
446  else
447  mSyncCheckRegister[s] = 0;
448  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;
449  // check the register content for the SYNC pattern
450  // Patterh is : 1100.1100.1100.1100.1111.0000.1111.0000 = 0xCCCCF0F0
451  // alternative Pattern for first corrupted SYNC word : 0100.1100.1100.1100.1111.0000.1111.0000 = 4CCC.F0F0
452  if (mSyncCheckRegister[s] == 0xCCCCF0F0 or mSyncCheckRegister[s] == 0x4CCCF0F0 or mSyncCheckRegister[s] == 0x0CCCF0F0) {
453  syncArray[s].setPos(mPacketNum, mFrameNum, mFilePos, h);
454  };
455  };

485 inline void RawReaderCRU::GBTFrame::getFrameHalfWords()
486 {
487  uint32_t P[5][4] = { { 19, 18, 17, 16 }, { 39, 38, 37, 36 }, { 63, 62, 61, 60 }, { 83, 82, 81, 80 }, { 107, 106, 105, 104 } };
488  uint32_t res = 0;
489  // i = Stream, j = Halfword
490  for (int i = 0; i < 5; i++)
491  for (int j = 0; j < 4; j++) {
492  res = bit(P[i][j], 4) | bit(P[i][j] - 4, 3) | bit(P[i][j] - 8, 2) | bit(P[i][j] - 12, 1) | bit(P[i][j] - 16, 0);
493  mFrameHalfWords[i][j] = res;
494  // std::cout << " S : " << i << " H : " << j << " : " << std::hex << res << std::endl;
495  };

509 inline void RawReaderCRU::GBTFrame::getAdcValues(ADCRawData& rawData)
510 {
511  uint32_t pos;
512  uint32_t v0;
513  uint32_t v1;
514  // loop over all the 5 data streams
515  for (int s = 0; s < 5; s++) {
516  // assemble the 2 ADC words from 4 half-words. Which halfwords are
517  // used depends on the position of the SYNC pattern.
518  pos = mSyncPos[s].getHalfWordPosition();
519  if (pos == 0) {
520  v0 = (mPrevFrameHalfWords[s][2] << 5) | mPrevFrameHalfWords[s][1];
521  v1 = (mFrameHalfWords[s][0] << 5) | mPrevFrameHalfWords[s][3];
522  } else if (pos == 1) {
523  v0 = (mPrevFrameHalfWords[s][3] << 5) | mPrevFrameHalfWords[s][2];
524  v1 = (mFrameHalfWords[s][1] << 5) | mFrameHalfWords[s][0];
525  } else if (pos == 2) {
526  v0 = (mFrameHalfWords[s][0] << 5) | mPrevFrameHalfWords[s][3];
527  v1 = (mFrameHalfWords[s][2] << 5) | mFrameHalfWords[s][1];
528  } else if (pos == 3) {
529  v0 = (mFrameHalfWords[s][1] << 5) | mFrameHalfWords[s][0];
530  v1 = (mFrameHalfWords[s][3] << 5) | mFrameHalfWords[s][2];
531  };
532  // add the two ADC values for the current processed stream
533  if (mSyncPos[s].synched() == true) {
534  // std::cout << "Adding ADC" << std::endl;
535  rawData.add(s, v0, v1);
536  };
537  };
